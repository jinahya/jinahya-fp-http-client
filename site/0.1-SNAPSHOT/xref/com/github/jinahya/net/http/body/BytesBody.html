18  package com.github.jinahya.net.http.body;
19  
20  
21  import com.github.jinahya.net.http.HeaderUtilities;
22  import java.io.ByteArrayOutputStream;
23  import java.io.IOException;
24  import java.io.InputStream;
25  import java.io.OutputStream;
26  import java.net.HttpURLConnection;
27  
28  
29  /**
30   *
31   * @author Jin Kwon &lt;jinahya_at_gmail.com&gt;
32   */
33  public class BytesBody extends BidirectionalBody<byte[]> {
34  
35  
36      public BytesBody(final byte[] value) {
37  
38          super();
39  
40          setValue(value);
41      }
42  
43  
44      public BytesBody() {
45  
46          this(null);
47      }
48  
49  
50      @Override
51      public void read(final HttpURLConnection connection) throws IOException {
52  
53          final ByteArrayOutputStream baos;
54          final Long contentLength
55              = HeaderUtilities.getResponseContentLength(connection);
56          if (contentLength != null) {
57              baos = new ByteArrayOutputStream(contentLength.intValue());
58          } else {
59              baos = new ByteArrayOutputStream();
60          }
61  
62          final InputStream input = connection.getInputStream();
63          try {
64              final byte[] buffer = new byte[1024];
65              for (int read = -1; (read = input.read(buffer)) != -1;
66                   baos.write(buffer, 0, read)) {
67              }
68              baos.flush();
69              setValue(baos.toByteArray());
70          } finally {
71              input.close();
72          }
73      }
74  
75  
76      @Override
77      public void write(final HttpURLConnection connection) throws IOException {
78  
79          final OutputStream output = connection.getOutputStream();
80          try {
81              output.write(getValue());
82              output.flush();
83          } finally {
84              output.close();
85          }
86      }
87  
88  
89  }
